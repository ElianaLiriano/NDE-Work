import numpy as np
import matplotlib.pyplot as plt


def derivativeplot(f,a,b,n,h):
    x = np.linspace(a,b,n)
    y = f(x)
    dy = ( f(x + h)- f(x - h)) / (2 * h)
    ddy = (f(x - h) -2 * f(x) + f(x + h)) / (h**2)
    
# Plotting 

    fig, axs = plt.subplots(3, 1, figsize=(8, 8), sharex = True )

    axs[0].plot(x,y)
    axs[0].set_ylabel('f(x)')

    axs[1].plot(x,dy)
    axs[1].set_ylabel('f \'(x)')

    axs[2].plot(x,ddy)
    axs[2].set_ylabel('f \'\'(x)')

    plt.xlabel('x')
    plt.show


def f1(x):
    return np.sin(x)

def f2(x):
    return x**3

f1 = derivativeplot(f1, 0, 2*np.pi, 1000, 0.1)
f2 = derivativeplot(f2, -1, 1, 1000, 0.1)

def integral(f,a,b,n):
    

#Computing Integral Length 
    h = (b-a)/(n-1)

    
#Computing Trapazoid Sum Rule 
    T = 0 
    for i in range(1,n):
        x1 = a + (i-1)*h
        x2 = a + i * h
        T += (f(x1) + f(x2)) * h/2

        
# Computing Simpson Sum Rule  
    S = 0 
    h_s = (b-a)/n
    
    for i in range(n):
        
        L = xi = a + (i)*h_s
        R= xi_1 = a + (i+1)*h_s
        M = (L+R)/2
        
        S += (f(L)+ 4 * f(M) + f(R)) * h_s/6
        
    return T,S
