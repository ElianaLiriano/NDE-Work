import numpy as np
import scipy.linalg as la
import numpy.linalg as LA
np.set_printoptions(formatter={'float': '{: 0.2f}'.format})

T = np.zeros((12,6))

for j in range(12):
    #Column 1
    n = j + 2
    T[j,0] = n
    
    A = la.hilbert(n)
    b = A[:,0] + A[:,n-1]
    x = np.zeros(n)
    x[0] = 1
    x[-1]= 1
    

    #Column 2
    T[j,1] = LA.cond(A) *1e-16
    
    
    #Column 3
    xb = LA.solve(A,b)
    T[j,2] = LA.norm(x-xb,np.inf)/LA.norm(x,np.inf)
    
    
    #Column 4 
    P,L,U = la.lu(A)
    y = la.solve(L,P. transpose() @ b)
    xLu = la.solve(U,y)
    T[j,3] = LA.norm(x-xLu,np.inf)/LA.norm(x,np.inf)
    
    
    #Column 5 
    
    L = LA.cholesky(A)
    y = la.solve(L,b)
    xchol= la.solve(L.T,y)
    T[j,4] = LA.norm(x-xchol,np.inf)/LA.norm(x,np.inf)
    
    
    #Column 6 
    bchol = A @ xchol
    T[j,5] = LA.norm(bchol-b,np.inf)
    
print(T)

import numpy as np  
import matplotlib.pyplot as plt 
import scipy

#1
duck = np.array([[0,1,1,1.5,2.5,3,3,4,2,2,1,-3.7,-5,-4,-3,0],[1,0,3.5,4,4,3.5,3,3,2,-1,-2,-2,1,0,1,1]])
t = np.linspace(0,2*np.pi,1000)
x = 6*np.sin(t)
y = 6*np.cos(t)
circle = np.array([x,y])

A = ([[1,1],[2,3]])
U,Σ,VT = scipy.linalg.svd(A)
Σ = np.identity(len(Σ))*Σ

# 2
newduck = np.dot(VT,duck)
newcircle = np.dot(VT,circle)


#3 
NWduck = np.dot(Σ,newduck)
NWcircle = np.dot(Σ,newcircle)

#4 
NEWduck = np.dot(U,NWduck)
NEWcircle = np.dot(U,NWcircle)


fig, axs = plt.subplots(nrows=2, ncols=2, constrained_layout=True)
fig.suptitle('SVD DUCKSSS')


axs[0,0].plot(duck[0,:],duck[1,:])
